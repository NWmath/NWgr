34template <typename _InputIterator1, typename _InputIterator2, typename _OutputIterator>
35_OutputIterator lower_bound_set_intersection(_InputIterator1 __first1, _InputIterator1 __last1, _InputIterator2 __first2,
36 _InputIterator2 __last2, _OutputIterator __result) {
37 while (__first1 != __last1 && __first2 != __last2)
38 if (*__first1 < *__first2) {
39 ++__first1;
40 __first1 = std::lower_bound(__first1, __last1, *__first2);
41 } else if (*__first2 < *__first1) {
42 ++__first2;
43 __first2 = std::lower_bound(__first2, __last2, *__first1);
44 } else {
45 *__result = *__first1;
46 ++__first1;
47 ++__first2;
48 ++__result;
49 }
50 return __result;
51}

53template <adjacency_list_graph GraphT>
54size_t triangle_count_vc(const GraphT& A) {
55 size_t triangles = 0;
56 counting_output_iterator counter(triangles);
57
58 size_t len = A.indices_.size() - 1;
59 auto ptrs = A.indices_;
60 auto idxs = std::get<0>(A.to_be_indexed_);
61
62 for (size_t i = 0; i < len; ++i) {
63 for (size_t j = ptrs[i]; j < ptrs[i + 1]; ++j) {
64 size_t k = idxs[j];
65 size_t k0 = ptrs[i];
66 size_t k1 = ptrs[i + 1];
67 size_t k2 = ptrs[k];
68 size_t k3 = ptrs[k + 1];
69
70 lower_bound_set_intersection(&idxs[k2], &idxs[k3], &idxs[k0], &idxs[k1], counter);
71 }
72 }
73
74 return triangles;
75}

78template <adjacency_list_graph GraphT>
79size_t triangle_count_v1(const GraphT& A) {
80 size_t triangles = 0;
81 auto first = A.begin();
82 auto last = A.end();
83
84 for (auto G = first; first != last; ++first) {
85 for (auto elt = (*first).begin(); elt != (*first).end(); ++elt) {
86 auto v = target(A, *elt);
87 triangles += nw::graph::intersection_size(elt, (*first).end(), G[v]);
88 }
89 }
90 return triangles;
91}

151 return triangle_count_async(threads, [&](std::size_t tid) {
152 std::size_t triangles = 0;
153 auto&& [u, ue] = nw::graph::block(last - first, threads, tid);
154 for (auto i = first + u, end = first + ue; i != end; ++i) {
155 for (auto j = (*i).begin(), end = (*i).end(); j != end; ++j) {
156 triangles += nw::graph::intersection_size(j, end, first[std::get<0>(*j)]);
157 }
158 }
159 return triangles;
160 });
