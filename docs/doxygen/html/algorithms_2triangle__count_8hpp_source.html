34template <adjacency_list_graph GraphT>
35size_t triangle_count_v0(const GraphT& A) {
36 size_t triangles = 0;
37 auto first = A.begin();
38 auto last = A.end();
39 for (auto u_neighors = first; u_neighors != last; ++u_neighors) {
40 for (auto elt = (*u_neighors).begin(); elt != (*u_neighors).end(); ++elt) {
41 auto v = target(A, *elt);
42 triangles += nw::graph::intersection_size(*u_neighors, A[v]);
43 }
44 }
45 return triangles;
46}

60template <class Op>
61std::size_t triangle_count_async(std::size_t threads, Op&& op) {
62 // Launch the workers.
63 std::vector<std::future<size_t>> futures(threads);
64 for (std::size_t tid = 0; tid < threads; ++tid) {
65 futures[tid] = std::async(std::launch::async, op, tid);
66 }
67
68 // Reduce the outcome.
69 int i = 0;
70 std::size_t triangles = 0;
71 for (auto&& f : futures) {
72 triangles += f.get();
73 }
74 return triangles;
75}
89template <typename RandomAccessIterator>
90[[gnu::noinline]] std::size_t triangle_count_v4(RandomAccessIterator first, RandomAccessIterator last, std::size_t threads = 1) {
91 return triangle_count_async(threads, [&](std::size_t tid) {
92 std::size_t triangles = 0;
93 for (auto i = first + tid; i < last; i += threads) {
94 for (auto j = (*i).begin(), end = (*i).end(); j != end; ++j) {
95 // assert(j < end);
96 triangles += nw::graph::intersection_size(j, end, first[std::get<0>(*j)]);
97 }
98 }
99 return triangles;
100 });
101}
