25template <typename Graph>
26class random_range {
27public:
28 random_range(Graph& g, size_t length = size_t(0xffffffffffffffffULL), vertex_id_t<Graph> first = 0, unsigned seed = 2049)
29 : the_graph_(g), starting_vertex(first), length_(length), distribution(0, 1.0), generator(seed),
30 dice(std::bind(distribution, generator)) {}
31
32 class edge_range_iterator {
33 private:
34 using vertex_id_type = vertex_id_t<Graph>;
35
36 random_range<Graph>& the_range_;
37 typename Graph::outer_iterator G;
38 vertex_id_type current_vertex;
39 size_t length_;
40
41 public:
42 edge_range_iterator(random_range<Graph>& range)
43 : the_range_(range), G(the_range_.the_graph_.begin()), current_vertex(the_range_.starting_vertex), length_(the_range_.length_) {}
44
45 edge_range_iterator& operator++() {
46 --length_;
47 typename Graph::inner_iterator u_begin = G[current_vertex].begin();
48 typename Graph::inner_iterator u_end = G[current_vertex].end();
49 size_t N = u_end - u_begin;
50 vertex_id_type pick = std::floor(N * the_range_.dice());
51
52 current_vertex = std::get<0>(u_begin[pick]);
53
54 return *this;
55 }
56
57 auto operator*() const { return current_vertex; }
58
59 struct end_sentinel_type {};
60
61 auto operator==(const end_sentinel_type&) const { return length_ == 0; }
62 bool operator!=(const end_sentinel_type&) const { return length_ != 0; }
63 };
64
65 typedef edge_range_iterator iterator;
66
67 auto begin() { return edge_range_iterator(*this); }
68 auto end() { return typename edge_range_iterator::end_sentinel_type(); }
69
70private:
71 Graph& the_graph_;
72
73 vertex_id_t<Graph> starting_vertex;
74 size_t length_;
75
76 std::uniform_real_distribution<double> distribution;
77 std::default_random_engine generator;
78
79 decltype(std::bind(distribution, generator)) dice;
80};
