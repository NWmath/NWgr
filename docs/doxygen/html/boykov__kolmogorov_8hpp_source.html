28const size_t INFINITE_D = 100000000;
29
30enum class tree_mem : bool { source = false, term = true };
31
32template <typename Graph>
33std::tuple<double, std::vector<tree_mem>> bk_maxflow(const Graph& A, std::vector<double>& cap) {
34 // std::clock_t start;
35 // double grow = 0;
36 // double augment = 0;
37 // double adopt = 0;
38 using vertex_id_type = vertex_id_t<Graph>;
39
40 size_t n_vtx = A.size();
41 vertex_id_type source = n_vtx;
42 vertex_id_type terminal = source + 1;
43 double max_flow = 0.0;
44 double tol = 1e-12;
45 size_t gtime = 0;
46
47 std::queue<vertex_id_type> active;
48 std::queue<vertex_id_type> orphans;
49 std::vector<tree_mem> tree_id(n_vtx);
50 std::vector<std::tuple<double*, double*>> trees(n_vtx);
51 std::vector<vertex_id_type> preds(n_vtx, null_vertex_v);
52 std::vector<size_t> timestamp(n_vtx);
53 std::vector<size_t> dist(n_vtx);
54
55 double * residual_cap, *back_residual_cap;
56 vertex_id_type connect_s;
57 vertex_id_type connect_t;
58 double min_cap;
59 auto G = A.begin();
60
61 for (size_t i = 0; i < n_vtx; ++i) {
62 if (cap[i] > 0) {
63 tree_id[i] = tree_mem::source;
64 preds[i] = source;
65 } else if (cap[i] < 0) {
66 max_flow += cap[i];
67 tree_id[i] = tree_mem::term;
68 preds[i] = terminal;
69 } else {
70 continue;
71 }
72
73 dist[i] = 1;
74 timestamp[i] = 0;
75 active.push(i);
76 }
77
78 while (true) {
79 bool found = false;
80 // start = std::clock();
81 // Grow
82 while (!active.empty()) {
83 auto p = active.front();
84
85 // This node was disconnected, so make it inactive
86 if (preds[p] == null_vertex_v) {
87 active.pop();
88 continue;
89 }
90
91 tree_mem t = tree_id[p];
92
93 for (auto it = G[p].begin(); it != G[p].end(); ++it) {
94 auto q = std::get<0>(*it);
95
96 if (t == tree_mem::term) {
97 // We are in the term tree, so we need to push flow "backwards", which
98 // is the residual capacity.
99 // residual_cap = std::get<2>(*it);
100 residual_cap = &std::get<1>(A.get_back_edge(p, q));
101 if (*residual_cap <= tol) {
102 // No residual capacity
103 continue;
104 }
105 back_residual_cap = &std::get<1>(*it);
106 // potential endpoint in the source tree
107 connect_s = q;
108 connect_t = p;
109 } else { // s-tree
110 residual_cap = &std::get<1>(*it);
111 if (*residual_cap <= tol) {
112 // no residual capacity
113 continue;
114 }
115 // back_residual_cap = std::get<2>(*it);
116 back_residual_cap = &std::get<1>(A.get_back_edge(p, q));
117 connect_s = p;
118 // potential endpoint in the terminal tree
119 connect_t = q;
120 }
121
122 // q is fresh (does not have a predecessor, not in the tree)
123 if (preds[q] == null_vertex_v) {
124 // Assign our tree to q
125 tree_id[q] = t;
126 // We are the predecessor now
127 preds[q] = p;
128 trees[q] = std::make_tuple(residual_cap, back_residual_cap);
129 active.push(q);
130 timestamp[q] = timestamp[p];
131 dist[q] = dist[p] + 1;
132 }
133 // Found a connection because q is in the other tree.
134 else if (tree_id[q] != t) {
135 found = true;
136 break;
137 }
138 // q is in the same tree
139 else if (timestamp[q] <= timestamp[p] && dist[q] > dist[p]) {
140 preds[q] = p;
141 trees[q] = std::make_tuple(residual_cap, back_residual_cap);
142 timestamp[q] = timestamp[p];
143 dist[q] = dist[p] + 1;
144 }
145 }
146
147 gtime++;
148
149 if (found) break;
150
151 active.pop();
152 }
153
154 if (!found) break;
155
156 // grow += (std::clock() - start);
157 // start = std::clock();
158 // Augment
159 min_cap = *residual_cap;
160 // Search the t-tree for min capacity
161 for (auto vtx = connect_t; vtx != terminal; vtx = preds[vtx]) {
162 if (preds[vtx] == terminal) {
163 min_cap = std::min(min_cap, -cap[vtx]);
164 } else {
165 auto edge = trees[vtx];
166 min_cap = std::min(min_cap, *std::get<0>(edge));
167 }
168 }
169 // Search the s-tree for min capacity
170 for (auto vtx = connect_s; vtx != source; vtx = preds[vtx]) {
171 if (preds[vtx] == source) {
172 min_cap = std::min(min_cap, cap[vtx]);
173 } else {
174 auto edge = trees[vtx];
175 min_cap = std::min(min_cap, *std::get<0>(edge));
176 }
177 }
178
179 max_flow += min_cap;
180 *residual_cap -= min_cap;
181 *back_residual_cap += min_cap;
182 // auto vtx = connect_t, next = preds[vtx]; vtx != terminal; vtx = next,
183 // next = preds[vtx]
184 // Update t-tree
185 for (auto vtx = connect_t, next = preds[vtx]; vtx != terminal; vtx = next) {
186 next = preds[vtx];
187 if (next == terminal) {
188 cap[vtx] += min_cap;
189 if (cap[vtx] >= 0) {
190 preds[vtx] = null_vertex_v;
191 orphans.push(vtx);
192 }
193 } else {
194 auto edge = trees[vtx];
195 *std::get<0>(edge) -= min_cap;
196 *std::get<1>(edge) += min_cap;
197 if (*std::get<0>(edge) <= 0) {
198 preds[vtx] = null_vertex_v;
199 orphans.push(vtx);
200 }
201 }
202 }
203 // Update s-tree
204 for (auto vtx = connect_s, next = preds[vtx]; vtx != source; vtx = next) {
205 next = preds[vtx];
206 if (next == source) {
207 cap[vtx] -= min_cap;
208 if (cap[vtx] <= 0) {
209 preds[vtx] = null_vertex_v;
210 orphans.push(vtx);
211 }
212 } else {
213 auto edge = trees[vtx];
214 *std::get<0>(edge) -= min_cap;
215 *std::get<1>(edge) += min_cap;
216 if (*std::get<0>(edge) <= 0) {
217 preds[vtx] = null_vertex_v;
218 orphans.push(vtx);
219 }
220 }
221 }
222
223 // augment += (std::clock() - start);
224 // start = std::clock();
225 // Adopt
226 while (!orphans.empty()) {
227 auto p = orphans.front();
228 orphans.pop();
229
230 tree_mem t = tree_id[p];
231 vertex_id_type min_pred{null_vertex_v};
232 size_t d_min = INFINITE_D;
233 std::tuple<double*, double*> min_edge;
234
235 for (auto it = G[p].begin(); it != G[p].end(); ++it) {
236 vertex_id_type q = std::get<0>(*it);
237 vertex_id_type target;
238 size_t d = 0;
239
240 if (tree_id[q] != t || preds[q] == p) continue;
241
242 if (t == tree_mem::term) {
243 residual_cap = &std::get<1>(*it);
244 if (*residual_cap <= tol) continue;
245 // back_residual_cap = std::get<2>(*it);
246 back_residual_cap = &std::get<1>(A.get_back_edge(p, q));
247 target = terminal;
248 } else {
249 // residual_cap = std::get<2>(*it);
250 residual_cap = &std::get<1>(A.get_back_edge(p, q));
251 if (*residual_cap <= tol) continue;
252 back_residual_cap = &std::get<1>(*it);
253 target = source;
254 }
255
256 for (auto vtx = q; vtx != target; vtx = preds[vtx]) {
257 if (timestamp[vtx] == gtime) {
258 d += dist[vtx];
259 break;
260 }
261
262 d++;
263
264 if (preds[vtx] == target) {
265 timestamp[vtx] = gtime;
266 dist[vtx] = 1;
267 break;
268 } else if (preds[vtx] == null_vertex_v) {
269 d = INFINITE_D;
270 break;
271 }
272 }
273
274 if (d < INFINITE_D) {
275 if (d < d_min) {
276 min_pred = q;
277 min_edge = std::make_tuple(residual_cap, back_residual_cap);
278 d_min = d;
279 }
280
281 for (auto vtx = q; vtx != target; vtx = preds[vtx]) {
282 if (timestamp[vtx] == gtime) break;
283
284 timestamp[vtx] = gtime;
285 dist[vtx] = d--;
286 }
287 }
288 }
289
290 // can't find new parent
291 if (min_pred == null_vertex_v) {
292 timestamp[p] = 0;
293
294 for (auto it = G[p].begin(); it != G[p].end(); ++it) {
295 auto q = std::get<0>(*it);
296
297 if (tree_id[q] != t) continue;
298
299 if (tree_id[q] == tree_mem::source && std::get<1>(*it) > tol) {
300 active.push(q);
301 } else if (tree_id[q] == tree_mem::term && std::get<1>(A.get_back_edge(p, q)) > tol) {
302 active.push(q);
303 }
304
305 if (preds[q] == p) {
306 orphans.push(q);
307 preds[q] = null_vertex_v;
308 }
309 }
310 }
311 // Found parent
312 else {
313 preds[p] = min_pred;
314 trees[p] = min_edge;
315 timestamp[p] = gtime;
316 dist[p] = d_min + 1;
317 }
318 }
319 // adopt += (std::clock() - start);
320 }
321 /*std::cout << "grow time: " << grow / (double) CLOCKS_PER_SEC <<
322 " augment time: " << augment / (double) CLOCKS_PER_SEC <<
323 " adopt time: " << adopt / (double) CLOCKS_PER_SEC << std::endl;*/
324 // std::cout << augment_ct << " aguments" << std::endl;
325 // std::cout << path_length*(1.0)/augment_ct << " avg path length" <<
326 // std::endl;
327 for (size_t i = 0; i < n_vtx; ++i) {
328 if (preds[i] == null_vertex_v) {
329 tree_id[i] = tree_mem::term;
330 }
331 }
332 return std::make_tuple(max_flow, tree_id);
333}
