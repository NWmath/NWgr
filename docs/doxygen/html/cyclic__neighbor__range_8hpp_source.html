62 struct iterator {
63 Iterator begin_;
64 Iterator i_;
65 difference_type stride_;
66
67 decltype(auto) operator*() { return std::make_tuple(static_cast<std::size_t>(i_ - begin_), *i_); }
68
69 iterator& operator++() {
70 i_ += stride_;
71 return *this;
72 }
73
74 bool operator!=(const iterator& rhs) const { return i_ != rhs.i_; }
75 };
76
78 iterator begin() const { return {begin_, begin_ + cycle_, stride_}; }
79
86 iterator end() const {
87 difference_type n = end_ - begin_ - cycle_; // shifted span for cycle
88 difference_type r = n % stride_; // remainder in last stride
89 difference_type e = (stride_ - r) % stride_; // amount past `end_` we'll go
90 return {begin_, end_ + e, stride_};
91 }
92
93 difference_type size() const {
94 // figure out the number of valid elements in our cycle, which depends on
95 // the total number of elements, where our cycle starts, and the stride
96 difference_type n = end_ - begin_;
97 return n / stride_ + ((cycle_ < n % stride_) ? 1 : 0);
98 }
99
100 bool empty() const { return size() == 0; }
101
106 bool is_divisible() const { return stride_ <= cutoff_; }
107};
