55template <class score_t, class accum_t, adjacency_list_graph Graph>
56bool BCVerifier(const Graph& g, std::vector<typename graph_traits<Graph>::vertex_id_type>& trial_sources,
57 std::vector<score_t>& scores_to_test) {
58 using vertex_id_type = typename graph_traits<Graph>::vertex_id_type;
59
60 std::vector<score_t> scores(num_vertices(g), 0);
61 for (auto& source : trial_sources) {
62 std::vector<int> depths(num_vertices(g), -1);
63 depths[source] = 0;
64 std::vector<accum_t> path_counts(num_vertices(g) + 1, 0);
65 path_counts[source] = 1;
66 std::vector<vertex_id_type> to_visit;
67 to_visit.reserve(num_vertices(g));
68 to_visit.push_back(source);
69 for (auto it = to_visit.begin(); it != to_visit.end(); it++) {
70 vertex_id_type u = *it;
71 for (auto edge : g[u]) {
72 vertex_id_type v = target(g, edge);
73 if (depths[v] == -1) {
74 depths[v] = depths[u] + 1;
75 to_visit.push_back(v);
76 }
77 if (depths[v] == depths[u] + 1) {
78 path_counts[v] += path_counts[u];
79 }
80 }
81 }
82
83 std::vector<std::vector<vertex_id_type>> verts_at_depth;
84 for (size_t i = 0; i < num_vertices(g); ++i) {
85 if (depths[i] != -1) {
86 if (depths[i] >= static_cast<int>(verts_at_depth.size())) {
87 verts_at_depth.resize(depths[i] + 1);
88 }
89 verts_at_depth[depths[i]].push_back(i);
90 }
91 }
92
93 std::vector<score_t> deltas(num_vertices(g), 0);
94 for (int depth = verts_at_depth.size() - 1; depth >= 0; depth--) {
95 for (vertex_id_type u : verts_at_depth[depth]) {
96 for (auto edge : g[u]) {
97 vertex_id_type v = target(g, edge);
98 if (depths[v] == depths[u] + 1) {
99 deltas[u] += static_cast<double>(path_counts[u]) / static_cast<double>(path_counts[v]) * (1 + deltas[v]);
100 }
101 }
102 scores[u] += deltas[u];
103 }
104 }
105 }
106
107 score_t biggest_score = *std::max_element(scores.begin(), scores.end());
108 for (size_t i = 0; i < num_vertices(g); ++i) {
109 scores[i] = scores[i] / biggest_score;
110 }
111
112 bool all_ok = true;
113
114 for (size_t i = 0; i < scores.size(); ++i) {
115 accum_t delta = abs(scores_to_test[i] - scores[i]);
116 if (delta > 1e-6) {
117 std::cout << i << ": " << scores[i] << " != " << scores_to_test[i] << " " << scores[i] - scores_to_test[i] << std::endl;
118 all_ok = false;
119 }
120 }
121
122 return all_ok;
123}

126template <adjacency_list_graph Graph, typename score_t = float, typename accum_t = size_t>
127std::vector<score_t> betweenness_brandes(const Graph& G, bool normalize = true) {
128 using vertex_id_type = typename Graph::vertex_id_type;
129
130 size_t n_vtx = num_vertices(G);
131 std::vector<score_t> centrality(n_vtx, 0);
132
133 std::stack<vertex_id_type> S;
134 std::queue<vertex_id_type> Q;
135 std::vector<accum_t> path_counts(n_vtx);
136 std::vector<score_t> d(n_vtx);
137
138 for (vertex_id_type s = 0; s < n_vtx; ++s) {
139 path_counts.assign(n_vtx, 0);
140 d.assign(n_vtx, -1);
141 std::vector<std::list<size_t>> P(n_vtx);
142
143 path_counts[s] = 1;
144 d[s] = 0;
145 Q.push(s);
146
147 while (!Q.empty()) {
148 auto v = Q.front();
149 Q.pop();
150 S.push(v);
151 for (auto inner = G[v].begin(); inner != G[v].end(); ++inner) {
152 auto w = target(G, *inner);
153 //auto w = std::get<0>(*inner);
154 if (d[w] < 0) {
155 Q.push(w);
156 d[w] = d[v] + 1;
157 }
158 if (d[w] == (d[v] + 1)) {
159 path_counts[w] += path_counts[v];
160 P[w].push_back(v);
161 }
162 }
163 }
164
165 std::vector<score_t> delta(n_vtx, 0);
166 while (!S.empty()) {
167 auto w = S.top();
168 S.pop();
169 for (auto it = P[w].begin(); it != P[w].end(); ++it) {
170 delta[*it] += static_cast<score_t>(path_counts[*it]) / static_cast<score_t>(path_counts[w]) * (1 + delta[w]);
171 }
172 if (w != s) {
173 centrality[w] += delta[w];
174 }
175 }
176 }
177 if (normalize) {
178 score_t largest = *std::max_element(centrality.begin(), centrality.end());
179 std::transform(centrality.begin(), centrality.end(), centrality.begin(), [&](auto &val) { return val /= largest; });
180 }
181 return centrality;
182}

184template <class score_t, class accum_t, adjacency_list_graph Graph, class OuterExecutionPolicy = std::execution::parallel_unsequenced_policy,
185 class InnerExecutionPolicy = std::execution::parallel_unsequenced_policy>
186auto bc2_v5(const Graph& graph, const std::vector<typename Graph::vertex_id_type>& sources, int threads,
187 OuterExecutionPolicy&& outer_policy = {}, InnerExecutionPolicy&& inner_policy = {}, bool normalize = true) {
188 using vertex_id_type = typename Graph::vertex_id_type;
189
190 vertex_id_type N = num_vertices(graph);
191 size_t M = graph.to_be_indexed_.size();
192 auto&& edges = std::get<0>(*(graph[0]).begin());
193 std::vector<score_t> bc(N);
194
195 const vertex_id_type num_bins = nw::graph::pow2(nw::graph::ceil_log2(threads));
196 const vertex_id_type bin_mask = num_bins - 1;
197
198 std::vector<std::future<void>> futures(sources.size());
199 for (size_t s_idx = 0; s_idx < sources.size(); ++s_idx) {
200 futures[s_idx] = std::async(
201 std::launch::async,
202 [&](vertex_id_type root) {
203 std::vector<vertex_id_type> levels(N);
204 nw::graph::AtomicBitVector succ(M);
205
206 // Initialize the levels to infinity.
207 std::fill(outer_policy, levels.begin(), levels.end(), std::numeric_limits<vertex_id_type>::max());
208
209 std::vector<accum_t> path_counts(N);
210 std::vector<tbb::concurrent_vector<vertex_id_type>> q1(num_bins);
211 std::vector<tbb::concurrent_vector<vertex_id_type>> q2(num_bins);
212 std::vector<std::vector<tbb::concurrent_vector<vertex_id_type>>> retired;
213
214 vertex_id_type lvl = 0;
215
216 path_counts[root] = 1;
217 q1[0].push_back(root);
218 levels[root] = lvl++;
219
220 bool done = false;
221 while (!done) {
222 std::for_each(outer_policy, q1.begin(), q1.end(), [&](auto&& q) {
223 std::for_each(inner_policy, q.begin(), q.end(), [&](auto&& u) {
224 for (auto&& elt : graph[u]) {
225 auto v = target(graph, elt);
226 auto&& infinity = std::numeric_limits<vertex_id_type>::max();
227 auto&& lvl_v = nw::graph::acquire(levels[v]);
228
229 // If this is our first encounter with this node, or
230 // it's on the right level, then propagate the counts
231 // from u to v, and mark the edge from u to v as used.
232 if (lvl_v == infinity || lvl_v == lvl) {
233 nw::graph::fetch_add(path_counts[v], nw::graph::acquire(path_counts[u]));
234 succ.atomic_set(&v - &edges); // edge(w,v) : P[w][v]
235 }
236
237 // We need to add v to the frontier exactly once the
238 // first time we encounter it, so we race to set its
239 // level and if we win that race we can be the one to
240 // add it.
241 if (lvl_v == infinity && nw::graph::cas(levels[v], infinity, lvl)) {
242 q2[u & bin_mask].push_back(v);
243 }
244 }
245 });
246 });
247
248 done = true;
249 for (size_t i = 0; i < num_bins; ++i) {
250 if (q2[i].size() != 0) {
251 done = false;
252 break;
253 }
254 }
255
256 retired.emplace_back(num_bins);
257 std::swap(q1, retired.back());
258 std::swap(q1, q2);
259
260 ++lvl;
261 }
262
263 std::vector<score_t> deltas(N);
264
265 std::for_each(retired.rbegin(), retired.rend(), [&](auto&& vvv) {
266 std::for_each(outer_policy, vvv.begin(), vvv.end(), [&](auto&& vv) {
267 std::for_each(inner_policy, vv.begin(), vv.end(), [&](auto&& u) {
268 score_t delta = 0;
269 for (auto&& elt : graph[u]) {
270 auto v = target(graph, elt);
271 if (succ.get(&v - &edges)) {
272 delta += path_counts[u] / path_counts[v] * (1.0f + deltas[v]);
273 }
274 }
275 nw::graph::fetch_add(bc[u], deltas[u] = delta);
276 });
277 });
278 });
279 },
280 sources[s_idx]);
281 }
282
283 for (auto&& f : futures) {
284 f.wait();
285 }
286
287 if (normalize) {
288 auto max = std::reduce(outer_policy, bc.begin(), bc.end(), 0.0f, nw::graph::max{});
289 std::for_each(outer_policy, bc.begin(), bc.end(), [&](auto&& j) { j /= max; });
290 }
291 return bc;
292}
