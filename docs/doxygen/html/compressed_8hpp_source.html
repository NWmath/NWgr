338 template <typename Comparator = decltype(std::less<index_t>{})>
339 void triangularize_(Comparator comp = std::less<index_t>{}) {
340 std::vector<index_t> new_indices_(indices_.size());
341 struct_of_arrays<Attributes...> new_to_be_indexed_(0);
342 new_to_be_indexed_.reserve(to_be_indexed_.size());
343
344 new_indices_[0] = 0;
345 for (size_t i = 0; i < N_; ++i) {
346 const auto begin = to_be_indexed_.begin() + indices_[i];
347 const auto end = to_be_indexed_.begin() + indices_[i + 1];
348 size_t k = 0;
349 for (auto j = begin; j != end; ++j) {
350 auto tmp = std::get<0>(*j);
351 if (comp(tmp, i)) {
352 new_to_be_indexed_.push_back(*j);
353 ++k;
354 }
355 }
356 new_indices_[i + 1] = new_indices_[i] + k;
357 }
358 indices_ = std::move(new_indices_);
359 to_be_indexed_ = std::move(new_to_be_indexed_);
360 }

375 /*
376 * Serial version to compute degree of each vertex.
377 * Use adjacent_difference to compute the degrees of each vertex:
378 * degs[0] = 0 after the computation hence
379 * we need to erase the first element of the vector
380 */
381 std::vector<index_t> degrees() const {
382 std::vector<index_t> degs(indices_.size());
383 std::adjacent_difference(indices_.begin(), indices_.end(), degs.begin());
384 degs.erase( degs.begin() );
385
386 if (g_debug_compressed) {
387 for (size_t i = 0, e = indices_.size() - 1; i < e; ++i)
388 assert(degs[i] == indices_[i + 1] - indices_[i]);
389 }
390 return degs;
391 }

393 /*
394 * Parallel version to compute degree of each vertex.
395 */
396 template <class ExecutionPolicy = std::execution::parallel_unsequenced_policy>
397 std::vector<index_t> degrees(ExecutionPolicy&& ex_policy = {}) const {
398 std::vector<index_t> degs(indices_.size() - 1);
399 tbb::parallel_for(tbb::blocked_range(0ul, indices_.size() - 1), [&](auto&& r) {
400 for (auto i = r.begin(), e = r.end(); i != e; ++i) {
401 degs[i] = indices_[i + 1] - indices_[i];
402 }
403 });
404 if (g_debug_compressed) {
405 for (size_t i = 0, e = indices_.size() - 1; i < e; ++i)
406 assert(degs[i] == indices_[i + 1] - indices_[i]);
407 }
408 return degs;

428 * Based on the new_id_perm of the vertices, relabel each vertex i into new_id_perm[i]
429 * and then sort each neighbor list.

431 template <class ExecutionPolicy = std::execution::parallel_unsequenced_policy>
432 void relabel_to_be_indexed(const std::vector<index_t>& new_id_perm, ExecutionPolicy&& ex_policy = {}) {
433 auto s = std::get<0>(to_be_indexed_).begin();
434 tbb::parallel_for(tbb::blocked_range(0ul, std::get<0>(to_be_indexed_).size()), [&](auto&& r) {

442 /*
443 * This function permutes the indices of the adjacency and to_be_indexed
444 * but does NOT relabel the ids in the to_be_indexed.
445 * */
446 template <class ExecutionPolicy = std::execution::parallel_unsequenced_policy>
447 std::vector<index_t> permute_by_degree(std::string direction = "descending", ExecutionPolicy&& ex_policy = {}) {
448 //1. get the degrees of all the vertices
449 size_t n = indices_.size() - 1;
450 std::vector degs = degrees<ExecutionPolicy>(ex_policy);
451 //2. populate permutation with vertex id
452 std::vector<index_t> perm(n);
453 tbb::parallel_for(tbb::blocked_range(0ul, n), [&](auto&& r) {
454 for (auto i = r.begin(), e = r.end(); i != e; ++i) {
455 perm[i] = i;
456 }
457 });
458
459 //3. do a proxy sort on the permutation based on the degree of each vertex
460 // in descending or ascending order
461 // this will permutate the vertex id in perm based on the degrees
462 if (direction == "descending") {
463 std::sort(ex_policy, perm.begin(), perm.end(), [&](auto a, auto b) { return degs[a] > degs[b]; });
464 } else if (direction == "ascending") {
465 std::sort(ex_policy, perm.begin(), perm.end(), [&](auto a, auto b) { return degs[a] < degs[b]; });
466 } else {
467 std::cout << "Unknown direction: " << direction << std::endl;
468 //return an empty perm array if unknown direction
469 return std::vector<index_t>{};
470 }
471
472 //4. allocate a vector for new_indices
473 std::vector<index_t> new_indices(indices_);
474 auto new_tmp = new_indices.begin() + 1;
475 std::vector<index_t> new_id_perm(n);
476
477 //5. permutate the old indices based on the degree of the new_id
478 // to get the new_id_perm
479 tbb::parallel_for(tbb::blocked_range(0ul, n), [&](auto&& r) {
480 for (auto old_id = r.begin(), e = r.end(); old_id != e; ++old_id) {
481 auto new_id = perm[old_id];
482 new_tmp[old_id] = degs[new_id];
483 new_id_perm[new_id] = old_id;
484 }
485 });
486
487 //6. Computes an inclusive prefix sum operation for the new_indices
488 // before the computation, new_indices stores the degree of each vertex (with new id)
489 std::inclusive_scan(ex_policy, new_indices.begin(), new_indices.end(), new_indices.begin());
490 //7. Permute each neighborhood of each vertex in to_be_indexed_ to their new place
491 // based on the new_id_perm
492 to_be_indexed_.permute(indices_, new_indices, new_id_perm);
493
494 //8. Overwrite the old indices_ with new_indices
495 indices_ = std::move(new_indices);
496
497 if (g_debug_compressed) {
498 auto newdegs = degrees();
499 for (size_t i = 0; i < n; ++i) {
500 //std::cout << i << ":" << newdegs[i] << std::endl;
501 assert(degs[i] == newdegs[new_id_perm[i]]);
502 }
503 stream_indices(std::cout);
504 }
505 return new_id_perm;
506 }

508 /*
509 * Permute the adjacency based on the degree of each vertex
510 * There are two major steps: 1. permute the indices_ and the to_be_indexed_
511 * 2. relabel the to_be_indexed_ if needed (which is not needed if it is part of bi-adjacency)
512 * WARNING:
513 * If sort_by_degree on a bi-adjacency, do NOT use sort_by_degree.
514 * Call permute_by_degree on adjacency<idx>,
515 * then call relabel_to_be_indexed on adjacency<(idx + 1) % 2>.
516 */
517 template <class ExecutionPolicy = std::execution::parallel_unsequenced_policy>
518 void sort_by_degree(std::string direction = "descending", ExecutionPolicy&& ex_policy = {}) {
519 auto&& perm = permute_by_degree(direction, ex_policy);
520 relabel_to_be_indexed(perm, ex_policy);
521 }
