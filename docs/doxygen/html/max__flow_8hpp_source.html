27static int DEFAULT_MAX = 1000;
28enum class default_dict { capacity_idx = 1, flow_idx = 2 };
29template <size_t Idx, typename Edge>
30auto backedge_property(Edge edge) {
31 return std::get<Idx>(edge);
32}
33//****************************************************************************
34template <typename Dict = default_dict, typename flowtype = double, typename Graph>
35flowtype max_flow(const Graph& A, vertex_id_type source, vertex_id_type sink, size_t max_iters = DEFAULT_MAX) {
36 struct tree_edge {
37 flowtype* capacity;
38 flowtype* flow;
39 flowtype* back_flow;
40 size_t predecessor;
41 };
42
43 size_t n_vtx = A.size();
44 std::vector<tree_edge> search_tree(n_vtx);
45 flowtype maxflow = 0;
46
47 for (size_t i = 0; i <= max_iters; ++i) {
48 auto mf_filter = [&A](vertex_id_type vtx, typename Graph::inner_iterator edge) {
49 return property<idx(Dict::flow_idx)>(edge) >=
50 property<idx(Dict::capacity_idx)>(edge) + backedge_property<idx(Dict::flow_idx)>(A.get_back_edge(vtx, edge));
51 };
52
53 filtered_bfs::filtered_bfs_edge_range ranges(A, source, sink, mf_filter);
54 auto ite = ranges.begin();
55 for (; ite != ranges.end(); ++ite) {
56 auto v = get_source(ite);
57 auto u = get_target(ite);
58 auto c = property_ptr<idx(Dict::capacity_idx)>(ite);
59 auto f = property_ptr<idx(Dict::flow_idx)>(ite);
60 search_tree[u].capacity = c;
61 search_tree[u].flow = f;
62 search_tree[u].predecessor = v;
63 search_tree[u].back_flow = &std::get<idx(Dict::flow_idx)>(A.get_back_edge(v, u));
64 }
65
66 if (!ite.found()) break;
67
68 flowtype min_cap = 100000;
69 size_t path_length = 0;
70 for (auto y : reverse_path(search_tree, sink, source)) {
71 min_cap = std::min(min_cap, *(search_tree[std::get<0>(y)].capacity) - *(search_tree[std::get<0>(y)].flow) +
72 *(search_tree[std::get<0>(y)].back_flow));
73 ++path_length;
74 }
75
76 maxflow += min_cap;
77 for (auto y : reverse_path(search_tree, sink, source)) {
78 *(search_tree[std::get<0>(y)].flow) += min_cap;
79 auto excess = *(search_tree[std::get<0>(y)].flow) - *(search_tree[std::get<0>(y)].capacity);
80 if (excess > 0) {
81 *(search_tree[std::get<0>(y)].flow) = *(search_tree[std::get<0>(y)].capacity);
82 *(search_tree[std::get<0>(y)].back_flow) -= excess;
83 }
84 }
85 }
86
87 return maxflow;
88}
