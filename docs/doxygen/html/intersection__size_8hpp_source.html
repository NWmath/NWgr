46template <class A, class B, class C, class D, class ExecutionPolicy>
47std::size_t intersection_size(A i, B&& ie, C j, D&& je, ExecutionPolicy&& ep) {
48 // Custom comparator because we know our iterator operator* produces tuples
49 // and we only care about the first value.
50 static constexpr auto lt = [](auto&& x, auto&& y) { return std::get<0>(x) < std::get<0>(y); };
51
52 // Use our own trivial loop for the intersection size when the execution
53 // policy is sequential, otherwise rely on std::set_intersection.
54 //
55 // @todo We really don't need set intersection. You'd hope that it would be
56 // efficient with the output counter, but it just isn't. Parallelizing
57 // the intersection size seems non-trivial though.
58 if constexpr (std::is_same_v<std::decay_t<ExecutionPolicy>, std::execution::sequenced_policy>) {
59 std::size_t n = 0;
60 while (i != ie && j != je) {
61 if (lt(*i, *j)) {
62 ++i;
63 } else if (lt(*j, *i)) {
64 ++j;
65 } else {
66 ++n;
67 ++i;
68 ++j;
69 }
70 }
71 return n;
72 (void)ep;
73 } else {
74 return std::set_intersection(std::forward<ExecutionPolicy>(ep), std::forward<A>(i), std::forward<B>(ie), std::forward<C>(j),
75 std::forward<D>(je), nw::graph::counter{}, lt);
76 }
77}
