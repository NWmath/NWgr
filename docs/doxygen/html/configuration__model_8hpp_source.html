25std::size_t power_gen (double dist_min, double dist_max) {
26 /* Inverse square. */
27 double exponent = -2;
28
29 double min_pow = pow (dist_min, exponent + 1);
30 double max_pow = pow (dist_max, exponent + 1);
31
32 return (std::size_t) pow((max_pow - min_pow) * drand48 ()
33 + min_pow, 1 / (exponent + 1));
34}
35static double standard_exponential_unlikely(bitgen_t *bitgen_state,
36 uint8_t idx, double x) {
37 if (idx == 0) {
38 /* Switch to 1.0 - U to avoid log(0.0), see GH 13361 */
39 return ziggurat_exp_r - log(1.0 - next_double(bitgen_state));
40 } else if ((fe_double[idx - 1] - fe_double[idx]) * next_double(bitgen_state) +
41 fe_double[idx] <
42 exp(-x)) {
43 return x;
44 } else {
45 return random_standard_exponential(bitgen_state);
46 }
47}
48double random_standard_exponential(bitgen_t *bitgen_state) {
49 uint64_t ri;
50 uint8_t idx;
51 double x;
52 ri = next_uint64(bitgen_state);
53 ri >>= 3;
54 idx = ri & 0xFF;
55 ri >>= 8;
56 x = ri * we_double[idx];
57 if (ri < ke_double[idx]) {
58 return x; /* 98.9% of the time we return here 1st try */
59 }
60 return standard_exponential_unlikely(bitgen_state, idx, x);
61}

64 return pow(1 - exp(-random_standard_exponential(bitgen_state)), 1. / a);

68/*
69* Return a random undirected bipartite graph (in edge_list) from two given degree sequences.
70* The bipartite graph is composed of two partitions.
71* By default, contiguous_id_space is false, set A has nodes 0 to
72* (deg_seqa.size() - 1) and set B has nodes 0 to (deg_seqb.size() - 1).
73* When contiguous_id_space is true, set A has nodes 0 to (deg_seqa.size() - 1)
74* and set B has nodes deg_seqa.size() to (deg_seqa.size() + deg_seqb.size() - 1).
75* Nodes from set A are connected to nodes in set B by choosing
76* randomly from the possible free stubs, one in A and one in B.
77* Note directed graph is not supported in configuration model.
78*
79*/
80template<class T>
81nw::graph::edge_list<nw::graph::undirected>
82configuration_model(std::vector<T>& deg_seqa, std::vector<T>& deg_seqb, bool contiguous_id_space = false) {
83 //validate degree sequences such that their summation are equivalent
84 nw::graph::edge_list<nw::graph::undirected> el;
85 T suma = std::reduce(std::execution::par_unseq, deg_seqa.cbegin(), deg_seqa.cend());
86 T sumb = std::reduce(std::execution::par_unseq, deg_seqb.cbegin(), deg_seqb.cend());
87 if (suma != sumb) {
88 std::cout << "Invalid degree sequences, sum(deg_seqa)!=sum(deg_seqb)" << std::endl;
89 return el;
90 }
91 auto populate_stubs = [&](auto& degree_sequence) -> std::vector<T> {
92 std::vector<T> stubs(suma);
93 std::size_t cursor = 0;
94 for (std::size_t v = 0, e = degree_sequence.size(); v < e; ++v) {
95 auto degree_v = degree_sequence[v];
96 auto begin = stubs.begin() + cursor;
97 cursor += degree_v;
98 auto end = begin + cursor;
99 std::fill(std::execution::par_unseq, begin, end, v);
100 }
101 return stubs;
102 };
103 auto astubs = populate_stubs(deg_seqa);
104 auto bstubs = populate_stubs(deg_seqb);
105
106 std::random_device rd;
107 std::mt19937 g(rd());
108
109 std::shuffle(astubs.begin(), astubs.end(), g);
110 std::shuffle(bstubs.begin(), bstubs.end(), g);
111
112 std::copy(astubs.begin(), astubs.end(), std::ostream_iterator<int>(std::cout, " "));
113
114
115 el.open_for_push_back();
116 for(std::size_t i = 0, e = astubs.size(); i < e; ++i) {
117 el.push_back(astubs[i], bstubs[i]);
118 }
119 el.close_for_push_back();
120 return el;
121}
