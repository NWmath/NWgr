237 bfs_edge_range2_iterator& operator++() {
238 auto& Q = the_range_.Q_;
239 auto& colors = the_range_.colors_;
240
241 Q.push({std::get<0>(*u_begin), size_t(0xffffffffffffffffULL)});
242 colors[std::get<0>(*u_begin)] = grey;
243
244 ++u_begin;
245 while (u_begin != u_end && colors[std::get<0>(*u_begin)] != white) {
246 ++u_begin;
247 }
248
249 while (u_begin == u_end) {
250 colors[v_] = black;
251
252 while (colors[std::get<0>(Q.top())] == black && !Q.empty())
253 Q.pop();
254
255 if (Q.empty()) break;
256
257 v_ = std::get<0>(Q.top());
258 u_begin = G[v_].begin();
259 u_end = G[v_].end();
260

262 ++u_begin;
263 }
264 }
265
266 return *this;
267 }

291template <typename Graph, typename Queue = std::queue<typename Graph::vertex_id_type>>
292class bfs_edge_range3 {
293private:
294 using vertex_id_type = typename Graph::vertex_id_type;
295
296public:
297 bfs_edge_range3(Graph& graph, vertex_id_type seed = 0) : the_graph_(graph), colors_(graph.end() - graph.begin(), white) {
298 Q_[0].push(seed);
299 colors_[seed] = grey;
300 //After adding seed, add the neighbors of seed to next queue
301 for (auto ite = the_graph_[seed].begin(); ite != the_graph_[seed].end(); ++ite) { // Explore neighbors
302 auto u = std::get<0>(*ite);
303 if (colors_[u] == white) {
304 colors_[u] = grey;
305 Q_[1].push(u);
306 }
307 }
308 }
309
310 bfs_edge_range3(const bfs_edge_range3&) = delete;
311 bfs_edge_range3(const bfs_edge_range3&&) = delete;
312
313 bool empty() { return Q_[0].empty(); }
314
315 class bfs_edge_range3_iterator {
316 public:
317 bfs_edge_range3_iterator(bfs_edge_range3<Graph>& range) : the_range_(range) {}
318
319 bfs_edge_range3_iterator& operator++() {
320 auto G = the_range_.the_graph_.begin();
321 Queue(&Q)[2] = the_range_.Q_;
322 auto& colors = the_range_.colors_;
323
324 if (!Q[1].empty()) { // if the neighbor queue is not empty
325 // we proceed all the neighbors
326 auto u = Q[1].front();
327 Q[1].pop(); // pop the child which has been visited
328 colors[u] = black; // mark the child as black (visited)
329 Q[0].push(u); // put the visited child to current frontier (it becomes parent)
330 while (Q[1].empty()) { // keep filling neighbors until the
331 // neighboring queue is not empty
332 if (Q[0].empty()) break; // break if both queue are empty
333
334 auto v = Q[0].front(); // get parent vertex v
335
336 for (auto ite = G[v].begin(); ite != G[v].end(); ++ite) { // Explore neighbors
337 u = std::get<0>(*ite);
338 if (colors[u] == white) {
339 colors[u] = grey;
340 Q[1].push(u); //add all the white neighbors of v
341 }
342 } //for
343 if (Q[1].empty()) { //if parent vertex v has no white neighbors, then v has done all traverse
344 Q[0].pop(); //pop it out
345 colors[v] = black; //be done with parent vertex v
346 }
347 // but we still need to populate the child queue, so we continue populate with the next parent vertex
348 } // while
349 } else {
350 auto v = Q[0].front();
351 Q[0].pop(); //if v has no neighbor, pop it
352 colors[v] = black;
353 }
354
355 return *this;
356 }
357
358 auto operator*() {
359 Queue(&Q)[2] = the_range_.Q_;
360 //if v has no neighbor, return a self loop
361 if (!Q[1].empty())
362 return std::make_tuple(Q[0].front(), Q[1].front());
363 else
364 return std::make_tuple(Q[0].front(), Q[0].front());
365 }
