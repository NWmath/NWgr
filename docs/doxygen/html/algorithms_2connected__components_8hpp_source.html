36// Verifies CC result by performing a BFS from a vertex in each component
37// - Asserts search does not reach a vertex with a different component label
38// - If the graph is directed, it performs the search as if it was undirected
39// - Asserts every vertex is visited (degree-0 vertex should have own label)
40template <adjacency_list_graph Graph, class Transpose, class Vector>
41static bool CCVerifier(const Graph& graph, Transpose&& xpose, Vector&& comp) {
42 using NodeID = typename nw::graph::vertex_id_t<std::decay_t<Graph>>;
43 std::unordered_map<NodeID, NodeID> label_to_source;
44 for (auto&& [n] : plain_range(graph)) {
45 label_to_source[comp[n]] = n;
46 }
47 std::vector<bool> visited(graph.size() + 1);
48 std::vector<NodeID> frontier;
49 frontier.reserve(graph.size() + 1);
50 for (auto&& [curr_label, source] : label_to_source) {
51 frontier.clear();
52 frontier.push_back(source);
53 visited[source] = true;
54 for (auto it = frontier.begin(); it != frontier.end(); it++) {
55 NodeID u = *it;
56 for (auto&& elt : graph[u]) {
57 auto v = target(graph, elt);
58 if (comp[v] != curr_label) {
59 return false;
60 }
61 if (!visited[v]) {
62 visited[v] = true;
63 frontier.push_back(v);
64 }
65 }
66 if (u < xpose.size()) {
67 for (auto&& elt : xpose[u]) {
68 auto v = target(xpose, elt);
69 if (comp[v] != curr_label) {
70 return false;
71 }
72 if (!visited[v]) {
73 visited[v] = true;
74 frontier.push_back(v);
75 }
76 }
77 }
78 }
79 }
80 NodeID i = 0;
81 for (auto&& visited : visited) {
82 if (!visited) {
83 //return false;
84 ++i;
85 }
86 }
87 if (0 < i) {
88 std::cout << "unvisited " << i << " " << graph.size() + 1 << " ";
89 return false;
90 }
91 return true;
92}

94template <typename Vector, typename T>
95static void link(T u, T v, Vector& comp) {
96 T p1 = nw::graph::acquire(comp[u]);
97 T p2 = comp[v];
98 while (p1 != p2) {
99 T high = std::max(p1, p2);
100 T low = p1 + (p2 - high);
101 T p_high = comp[high];
102
103 if ((p_high == low) || (p_high == high && comp[high].compare_exchange_strong(high, low))) break;
104 p1 = comp[p_high];
105 p2 = comp[low];
106 }
107}

109template <typename Execution, typename Vector>
110static void compress(Execution exec, Vector& comp) {
111 std::for_each(exec, counting_iterator(0ul), counting_iterator(comp.size()), [&](auto n) {
112 while (comp[n] != comp[comp[n]]) {
113 auto foo = nw::graph::acquire(comp[n]);
114 auto bar = nw::graph::acquire(comp[foo]);
115 nw::graph::release(comp[n], bar);
116 }
117 });
118}

120template <typename Vector, typename T>
121static T sample_frequent_element(const Vector& comp, size_t num_samples = 1024) {
122 std::unordered_map<T, int> counts(32);
123 std::mt19937 gen;
124 std::uniform_int_distribution<T> distribution(0, comp.size() - 1);
125
126 for (size_t i = 0; i < num_samples; ++i) {
127 T n = distribution(gen);
128 counts[comp[n]]++;
129 }
130
131 auto&& [num, count] = *std::max_element(counts.begin(), counts.end(), [](auto&& a, auto&& b) { return std::get<1>(a) < std::get<1>(b); });
132 float frac_of_graph = static_cast<float>(count) / num_samples;
133 std::cout << "Skipping largest intermediate component (ID: " << num << ", approx. " << int(frac_of_graph * 100) << "% of the graph)\n";
134 return num;
135}

137template <typename Execution, adjacency_list_graph Graph1, adjacency_list_graph Graph2>
138static auto afforest(Execution& exec, Graph1& graph, Graph2& t_graph, const size_t neighbor_rounds = 2) {
139 using vertex_id_type = vertex_id_t<Graph1>;
140 std::vector<std::atomic<vertex_id_type>> comp(graph.size() + 1);
141 std::for_each(exec, counting_iterator(0ul), counting_iterator(comp.size()), [&](vertex_id_type n) { comp[n] = n; });
142 for (size_t r = 0; r < neighbor_rounds; ++r) {
143 std::for_each(exec, counting_iterator(0ul), counting_iterator(comp.size()), [&](vertex_id_type u) {
144 if (r < graph[u].size()) {
145 link(u, std::get<0>(graph[u].begin()[r]), comp);
146 }
147 });
148 compress(exec, comp);
149 }
150
151 vertex_id_type c = sample_frequent_element<std::vector<std::atomic<vertex_id_type>>, vertex_id_type>(comp);
152
153 std::for_each(exec, counting_iterator(0ul), counting_iterator(comp.size()), [&](vertex_id_type u) {
154 if (comp[u] == c) return;
155
156 if (neighbor_rounds < graph[u].size()) {
157 for (auto v = graph[u].begin() + neighbor_rounds; v != graph[u].end(); ++v) {
158 link(u, std::get<0>(*v), comp);
159 }
160 }
161
162 if (t_graph.size() != 0) {
163 for (auto&& elt : t_graph[u]) {
164 auto v = target(t_graph, elt);
165 link(u, v, comp);
166 }
167 }
168 });
169
170 compress(exec, comp);
171
172 return comp;
173}
