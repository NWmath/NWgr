42template <adjacency_list_graph Graph, adjacency_list_graph GraphT>
43bool BFSVerifier(const Graph& g, GraphT& g_t, vertex_id_t<Graph> source, std::vector<vertex_id_t<Graph>>& parent) {
44 using vertex_id_type = vertex_id_t<Graph>;
45
46 std::vector<vertex_id_type> depth(num_vertices(g), std::numeric_limits<vertex_id_type>::max());
47 depth[source] = 0;
48 std::vector<vertex_id_type> to_visit;
49 to_visit.reserve(num_vertices(g));
50 to_visit.push_back(source);
51 auto out_neigh = g.begin();
52 auto in_neigh = g_t.begin();
53 for (auto it = to_visit.begin(); it != to_visit.end(); it++) {
54 vertex_id_type u = *it;
55 for (auto edge : out_neigh[u]) {
56 vertex_id_type v = target(g, edge);
57 //vertex_id_type v = std::get<0>(edge);
58 if (depth[v] == std::numeric_limits<vertex_id_type>::max()) {
59 depth[v] = depth[u] + 1;
60 to_visit.push_back(v);
61 }
62 }
63 }
64 for (vertex_id_type u = 0; u < num_vertices(g); ++u) {
65 if ((depth[u] != std::numeric_limits<vertex_id_type>::max()) && (parent[u] != std::numeric_limits<vertex_id_type>::max())) {
66 if (u == source) {
67 if (!((parent[u] == u) && (depth[u] == 0))) {
68 std::cout << "Source wrong " << u << " " << parent[u] << " " << depth[u] << std::endl;
69 return false;
70 }
71 continue;
72 }
73 bool parent_found = false;
74 for (auto edge : in_neigh[u]) {
75 vertex_id_type v = target(g_t, edge);
76 if (v == parent[u]) {
77 //if(it != out_neigh[v].end()) {
78 if (depth[v] != depth[u] - 1) {
79 std::cout << "Wrong depths for " << u << " & " << v << std::endl;
80 return false;
81 }
82 parent_found = true;
83 break;
84 }
85 }
86 if (!parent_found) {
87 std::cout << "Couldn't find edge from " << parent[u] << " to " << u << std::endl;
88 return false;
89 }
90 } else if (depth[u] != parent[u]) {
91 std::cout << "Reachability mismatch " << u << " " << depth[u] << " " << parent[u] << std::endl;
92 return false;
93 }
94 }
95 return true;
96}
106template <adjacency_list_graph Graph>
107auto bfs_v0(const Graph& graph, vertex_id_t<Graph> root) {
108 using vertex_id_type = vertex_id_t<Graph>;
109
110 std::deque<vertex_id_type> q1, q2;
111 std::vector<vertex_id_type> level(num_vertices(graph), std::numeric_limits<vertex_id_type>::max());
112 std::vector<vertex_id_type> parents(num_vertices(graph), std::numeric_limits<vertex_id_type>::max());
113 size_t lvl = 0;
114
115 q1.push_back(root);
116 level[root] = lvl++;
117 parents[root] = root;
118
119 while (!q1.empty()) {
120
121 std::for_each(q1.begin(), q1.end(), [&](vertex_id_type u) {
122 std::for_each(graph[u].begin(), graph[u].end(), [&](auto&& x) {
123 vertex_id_type v = target(graph, x);
124 if (level[v] == std::numeric_limits<vertex_id_type>::max()) {
125 q2.push_back(v);
126 level[v] = lvl;
127 parents[v] = u;
128 }
129 });
130 });
131 std::swap(q1, q2);
132 q2.clear();
133 ++lvl;
134 }
135 return parents;
136}

149template <adjacency_list_graph OutGraph, adjacency_list_graph InGraph>
150[[gnu::noinline]] auto bfs_v11(const OutGraph& out_graph, const InGraph& in_graph, vertex_id_t<OutGraph> root, int num_bins = 32,
151 int alpha = 15, int beta = 18) {
152
153 using vertex_id_type = vertex_id_t<OutGraph>;
154
155 const std::size_t n = nw::graph::pow2(nw::graph::ceil_log2(num_bins));
156 const std::size_t N = num_vertices(out_graph);
157 const std::size_t M = out_graph.to_be_indexed_.size();
158 std::vector<tbb::concurrent_vector<vertex_id_type>> q1(n), q2(n);
159
160 std::vector<vertex_id_type> parents(N);
161 nw::graph::AtomicBitVector front(N, false);
162 nw::graph::AtomicBitVector curr(N);
163
164 constexpr const auto null_vertex = null_vertex_v<vertex_id_type>();
165 std::fill(std::execution::par_unseq, parents.begin(), parents.end(), null_vertex);
166
167 std::uint64_t edges_to_check = M;
168 std::uint64_t scout_count = out_graph[root].size();
169
170 parents[root] = root;
171 q1[root % n].push_back(root);
172
173 bool done = false;
174 while (!done) {
175 if (scout_count > edges_to_check / alpha) {
176 std::size_t awake_count = 0;
177 // Initialize the frontier bitmap from the frontier queues, and count the
178 // number of non-zeros.
179 /*
180 std::size_t awake_count = nw::graph::parallel_for(
181 tbb::blocked_range(0ul, q1.size()),
182 [&](auto&& i) {
183 auto&& q = q1[i];
184 std::for_each(q.begin(), q.end(), [&](auto&& u) { curr.atomic_set(u); });
185 return q.size();
186 }, std::plus{}, 0ul);
187 */
188 std::for_each(std::execution::par_unseq, q1.begin(), q1.end(), [&](auto&& q) {
189 nw::graph::fetch_add(awake_count, q.size());
190 std::for_each(std::execution::par_unseq, q.begin(), q.end(), [&](auto&& u) { curr.atomic_set(u); });
191 });
192
193 std::size_t old_awake_count = 0;
194 do {
195 old_awake_count = awake_count;
196 std::swap(front, curr);
197 curr.clear();
198
199 awake_count = tbb::parallel_reduce(
200 tbb::blocked_range(0ul, N), 0ul,
201 [&](auto&& range, auto count) {
202 for (auto&& u = range.begin(), e = range.end(); u != e; ++u) {
203 if (null_vertex == parents[u]) {
204 for (auto&& elt : in_graph[u]) {
205 auto v = target(in_graph, elt);
206 if (front.get(v)) {
207 curr.atomic_set(u);
208 parents[u] = v;
209 ++count;
210 break;
211 }
212 }
213 }
214 }
215 return count;
216 },
217 std::plus{});
218 } while ((awake_count >= old_awake_count) || (awake_count > N / beta));
219
220 if (awake_count == 0) {
221 return parents;
222 }
223
224 tbb::parallel_for(curr.non_zeros(nw::graph::pow2(15)), [&](auto&& range) {
225 for (auto &&i = range.begin(), e = range.end(); i != e; ++i) {
226 q2[*i % n].push_back(*i);
227 }
228 });
229
230 scout_count = 1;
231 } else {
232 edges_to_check -= scout_count;
233 /*
234 scout_count = nw::graph::parallel_for(
235 tbb::blocked_range(0ul, q1.size()),
236 [&](auto&& i) {
237 auto&& q = q1[i];
238 return nw::graph::parallel_for(
239 tbb::blocked_range(0ul, q.size()),
240 [&](auto&& i) {
241 auto&& u = q[i];
242 return nw::graph::parallel_for(
243 out_graph[u],
244 [&](auto&& v) {
245 auto curr_val = parents[v];
246 if (null_vertex == curr_val) {
247 if (nw::graph::cas(parents[v], curr_val, u)) {
248 q2[u % n].push_back(v);
249 return out_graph[v].size();
250 }
251 }
252 return 0ul;
253 },
254 std::plus{}, 0ul);
255 },
256 std::plus{}, 0ul);
257 },
258 std::plus{}, 0ul);
259 */
260 scout_count = nw::graph::parallel_reduce(
261 tbb::blocked_range(0ul, q1.size()),
262 [&](auto&& i) {
263 auto&& q = q1[i];
264 return nw::graph::parallel_reduce(
265 tbb::blocked_range(0ul, q.size()),
266 [&](auto&& i) {
267 size_t count = 0;
268 auto&& u = q[i];
269 for (auto&& elt : out_graph[u]) {
270 auto v = target(out_graph, elt);
271 auto curr_val = parents[v];
272 if (null_vertex == curr_val) {
273 if (nw::graph::cas(parents[v], curr_val, u)) {
274 q2[u % n].push_back(v);
275 count += out_graph[v].size();
276 }
277 }
278 }
279 return count;
280 }, std::plus{}, 0ul);
281 }, std::plus{}, 0ul);
282 }
283
284 done = true;
285 for (auto&& q : q2) {
286 if (q.size() != 0) {
287 done = false;
288 break;
289 }
290 }
291 std::swap(q1, q2);
292 for (auto&& q : q2) {
293 q.clear();
294 }
295
296 }
297
298 return parents;
299}
