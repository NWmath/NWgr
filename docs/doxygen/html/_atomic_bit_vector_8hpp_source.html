144 class non_zero_iterator {
145 Word* data_;
146 std::size_t n_;
147 std::size_t word_;
148 std::size_t bit_ = 0;
149
150 public:
151 using value_type = std::size_t;
152 using difference_type = std::ptrdiff_t;
153 using reference = const std::size_t; // need something better
154 using pointer = void;
155 using iterator_category = std::input_iterator_tag;

157 non_zero_iterator(Word* data, std::size_t n, std::size_t word) : data_(data), n_(n), word_(word) { search_next(); }
158
159 reference operator*() { return word_ * BITS + bit_; }

165 non_zero_iterator& operator++() {
166 // if there is possibly another non-zero in the current word then use it
167 if (word_ < n_ && ++bit_ < BITS) {
168 if (Word w = data_[word_] >> bit_) {
169 bit_ += __builtin_ctzl(w);
170 return *this;
171 }
172 }
173
174 // otherwise we're on a slow path because we have to search for the next
175 // non-zero word
176 ++word_;
177 return search_next();
178 }
179
180 non_zero_iterator operator++(int) {
181 non_zero_iterator i(*this);
182 ++(*this);
183 return i;
184 }
185
186 bool operator!=(const non_zero_iterator& b) const { return (word_ != b.word_ || bit_ != b.bit_ || data_ != b.data_); }
187
188 bool operator==(const non_zero_iterator& b) const { return !operator!=(b); }
189
190 private:
191 non_zero_iterator& search_next() {
192 // search for the next non-zero word, starting with word_
193 for (; word_ < n_; ++word_) {
194 if (Word w = data_[word_]) {
195 bit_ = __builtin_ctzl(w);
196 return *this;
197 }
198 }
199
200 // saturate for comparisons with end()
201 bit_ = 0;
202 word_ = n_;
203 return *this;
204 }
205 };

241 static constexpr std::tuple<std::size_t, Word> split(std::size_t i) {
242 std::size_t word = i / BITS;
243 std::size_t offset = i % BITS;
244 Word mask = Word(1) << offset;
245 return {word, mask};
246 }

248 static constexpr std::size_t words(std::size_t n) { return n / BITS + ((n % BITS) ? 1 : 0); }
