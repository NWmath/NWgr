26template <class Iterator>
27class cyclic_range_adapter {

34 Iterator begin_;

41 template <class Range, class Cutoff>
42 cyclic_range_adapter(Range&& range, Cutoff cutoff)
43 : begin_(range.begin()), end_(range.end()), cutoff_(nw::graph::pow2(nw::graph::ceil_log2(cutoff))) {}

48 cyclic_range_adapter(cyclic_range_adapter& rhs, tbb::split)
49 : begin_(rhs.begin_), end_(rhs.end_), cutoff_(rhs.cutoff_), cycle_(rhs.cycle_ + rhs.stride_), stride_(rhs.stride_ *= 2) {}

51 struct iterator {
52 Iterator i_;
53 difference_type stride_;
54
55 decltype(auto) operator*() { return *i_; }
56
57 iterator& operator++() {
58 i_ += stride_;
59 return *this;
60 }
61
62 bool operator!=(const iterator& rhs) const { return i_ != rhs.i_; }
63 };
64
66 iterator begin() const { return {begin_ + cycle_, stride_}; }
67
74 iterator end() const {
75 difference_type n = end_ - begin_ - cycle_; // shifted span for cycle
76 difference_type r = n % stride_; // remainder in last stride
77 difference_type e = (stride_ - r) % stride_; // amount past `end_` we'll go
78 return {end_ + e, stride_};
79 }
80
81 difference_type size() const {
82 // figure out the number of valid elements in our cycle, which depends on
83 // the total number of elements, where our cycle starts, and the stride
84 difference_type n = end_ - begin_;
85 return n / stride_ + ((cycle_ < n % stride_) ? 1 : 0);
86 }
87
88 bool empty() const { return size() == 0; }
89
94 bool is_divisible() const { return stride_ <= cutoff_; }
95};
