21template<class T>
22class counting_iterator {
23private:
24 T my_counter_;
25public:
26 using iterator_category = std::random_access_iterator_tag;
27 using value_type = T;
28 using difference_type = std::make_signed<T>::type;
29 using reference = value_type&;
30 using pointer = value_type*;
31
32 counting_iterator() : my_counter_() {}
33 explicit counting_iterator(T init) : my_counter_(init) {}
34
35 reference operator*() { return my_counter_; }
36 value_type operator[](difference_type index) const { return *(*this + index); }
37
38 difference_type operator-(const counting_iterator& it) const {return my_counter_ - it.my_counter_; }
39
40 counting_iterator& operator+=(difference_type forward) { my_counter_ += forward; return *this; }
41 counting_iterator& operator-=(difference_type backward) { my_counter_ -= backward; return *this; }
42 counting_iterator& operator++() { return *this += 1; }
43 counting_iterator& operator--() { return *this -= 1; }
44
45 counting_iterator operator++(int) {

47 ++(*this);
48 return it;

51 counting_iterator operator--(int) {
52 counting_iterator it(*this);
53 --(*this);
54 return it;
55 }
56
57 counting_iterator operator+(difference_type forward) const { return counting_iterator(my_counter_ + forward); }
58 counting_iterator operator-(difference_type backward) const { return counting_iterator(my_counter_ - backward); }
59 friend counting_iterator operator+(difference_type forward, const counting_iterator it) { return it + forward; }
60
61 bool operator==(const counting_iterator& it) const { return 0 == (*this - it); }
62 bool operator!=(const counting_iterator& it) const { return 0 != (*this - it); }
63 bool operator<(const counting_iterator& it) const { return 0 > (*this - it); }

65 bool operator<=(const counting_iterator& it) const { return 0 >= (*this - it); }
66 bool operator>=(const counting_iterator& it) const { return 0 <= (*this - it); }
67}; // counting_iterator
