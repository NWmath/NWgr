28enum three_colors { black, white, grey };
29
30template <typename Graph, typename Queue = std::queue<vertex_id_t<Graph>>, typename Filter = std::function<bool()>>
31class filtered_bfs_edge_range {
32
33public:
34 using vertex_id_type = vertex_id_t<Graph>;
35
36 filtered_bfs_edge_range(
37 Graph& graph, vertex_id_type source, vertex_id_type target,
38 Filter filter = [](vertex_id_type v, typename Graph::inner_iterator iter) { return false; })
39 : the_graph_(graph), colors_(graph.end() - graph.begin(), white), target_(target), filter_(filter) {
40 Q_.push(source);
41 colors_[source] = grey;
42 }
43
44 filtered_bfs_edge_range(const filtered_bfs_edge_range&) = delete;
45 filtered_bfs_edge_range(const filtered_bfs_edge_range&&) = delete;
46
47 bool empty() { return Q_.empty(); }
48 bool found() { return colors_[target_] == grey; }
49 bool done() { return (found() || unreachable); }
50 bool unreachable = false;
51
52 class filtered_bfs_edge_range_iterator {
53
54 public:
55 filtered_bfs_edge_range_iterator(filtered_bfs_edge_range<Graph, Queue, Filter>& range)
56 : the_range_(range), G(the_range_.the_graph_.begin()), v_(the_range_.Q_.front()), u_begin(G[v_].begin()),
57 iterator_target_(the_range_.target_), filter_(the_range_.filter_) {
58
59 while (filter_(v_, u_begin)) {
60 ++u_begin;
61 if (u_begin == G[v_].end()) {
62 the_range_.unreachable = true;
63 break;
64 }
65 }
66 }
67
68 filtered_bfs_edge_range_iterator(const filtered_bfs_edge_range_iterator& ite)
69 : the_range_(ite.the_range_), G(ite.G), v_(ite.v_), u_begin(u_begin), filter_(ite.filter_) {}
70
71 filtered_bfs_edge_range_iterator& operator++() {
72 auto& Q = the_range_.Q_;
73 auto& colors = the_range_.colors_;
74
75 Q.push(std::get<0>(*u_begin));
76 colors[std::get<0>(*u_begin)] = grey;
77
78 if (iterator_target_ == std::get<0>(*u_begin)) {
79 colors[std::get<0>(*u_begin)] = grey;
80 return *this;
81 }
82 ++u_begin;
83
84 while (u_begin != G[v_].end()) {
85 if (colors[std::get<0>(*u_begin)] == white && !filter_(v_, u_begin)) break;
86
87 ++u_begin;
88 }
89
90 while (u_begin == G[v_].end()) {
91 colors[v_] = black;
92 Q.pop();
93 if (Q.empty()) {
94 the_range_.unreachable = true;
95 break;
96 }
97
98 v_ = Q.front();
99 assert(colors[v_] == grey);
100
101 u_begin = G[v_].begin();
102
103 while (u_begin != G[v_].end()) {
104 if (colors[std::get<0>(*u_begin)] == white && !filter_(v_, u_begin)) break;
105
106 ++u_begin;
107 }
108 }
109 return *this;
110 }
111
112 auto operator*() { return *u_begin; }
113 class end_sentinel_type {
114 public:
115 end_sentinel_type() {}
116 };
117
118 auto get_index() { return v_; };
119
120 auto operator==(const end_sentinel_type&) const { return the_range_.done(); }
121 bool operator!=(const end_sentinel_type&) const { return !the_range_.done(); }
122
123 bool found() { return the_range_.found(); }
124
125 private:
126 filtered_bfs_edge_range<Graph, Queue, Filter>& the_range_;
127 typename Graph::outer_iterator G;
128 vertex_id_type v_;
129 typename Graph::inner_iterator u_begin;
130 vertex_id_type iterator_target_;
131 Filter filter_;
132 };
133

144 Filter filter_;
145};
