24enum ready_to_process { yes, no };
25
26template <typename Graph, typename Queue = std::queue<vertex_id_t<Graph>>>
27class dag_range {
28
29public:
30 using vertex_id_type = vertex_id_t<Graph>;
31
32 typedef std::vector<std::vector<size_t>> VOV;
33
34 dag_range(Graph& graph, VOV pred_list, VOV successor_list)
35 : the_graph_(graph), _predecessor_list(pred_list), _successor_list(successor_list) {
36 /*Set all pred done counter to zero*/
37 _pred_done_counter.resize(the_graph_.size(), 0);
38 /*Insert all the roots in the queue*/
39 for (size_t i = 0; i < the_graph_.size(); i++) {
40 if (pred_list[i].size() == 0) {
41 Q_.push(i);
42 }
43 }
44 }
45 dag_range(const dag_range&) = delete;
46 dag_range(const dag_range&&) = delete;
47
48 bool empty() { return Q_.empty(); }
49
50 class dag_range_iterator {
51 private:
52 dag_range<Graph, Queue>& the_range_;
53 vertex_id_type v_;
54 std::vector<size_t>::iterator successor_iterator;
55
56 public:
57 dag_range_iterator(dag_range<Graph>& range)
58 : the_range_(range), v_(the_range_.Q_.front()), successor_iterator(the_range_._successor_list[v_].begin()) {
59 advance_next();
60 }
61
62 void advance_next() {
63 auto& Q = the_range_.Q_;
64 /*Check whether we have processed all the successors of the current vertex*/
65 if (successor_iterator == the_range_._successor_list[v_].end()) {
66 Q.pop(); /*remove current vertex*/
67 while (!Q.empty()) {
68 v_ = Q.front(); /*If not empty, get the next vertex*/
69 /*If no successor, continue for the next*/
70 if (the_range_._successor_list[v_].size() == 0) {
71 Q.pop();
72 continue;
73 }
74 /*We have to make sure that the last vertex in the queue*/
75 if (the_range_._successor_list[v_].size() > 0) {
76 /*Reset the iterator for the successor list*/
77 successor_iterator = the_range_._successor_list[v_].begin();
78 break;
79 }
80 }
81 }
82 if (the_range_._successor_list[v_].size() > 0) {
83 the_range_._ready_to_process = no;
84 the_range_._pred_done_counter[*successor_iterator] += 1;
85 if (the_range_._pred_done_counter[*successor_iterator] == the_range_._predecessor_list[*successor_iterator].size()) {
86 the_range_._ready_to_process = yes;
87 Q.push(*successor_iterator);
88 }
89 }
90 }
91
92 dag_range_iterator& operator++() {
93 successor_iterator++;
94 advance_next();
95 return *this;
96 }
97
98 auto operator*() {
99 return std::tuple<vertex_id_type, vertex_id_type, ready_to_process>(v_, *successor_iterator, the_range_._ready_to_process);
100 }
101 class end_sentinel_type {
102 public:
103 end_sentinel_type() {}
104 };
105 auto operator==(const end_sentinel_type&) const { return the_range_.empty(); }
106 bool operator!=(const end_sentinel_type&) const { return !the_range_.empty(); }
107 };
108 auto begin() { return dag_range_iterator(*this); }
109 auto end() { return typename dag_range_iterator::end_sentinel_type(); }
110
111private:
112 Graph& the_graph_;
113 Queue Q_;
114 VOV _predecessor_list;
115 VOV _successor_list;
116 std::vector<uint64_t> _pred_done_counter;
117 ready_to_process _ready_to_process;
118};
